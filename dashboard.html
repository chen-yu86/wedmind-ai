async function sendMessage(){ const msg = document.getElementById("messageInput").value; if(!msg) return; const chatBox = document.getElementById("chatBox"); const userMsg = document.createElement("div"); userMsg.className="message user"; userMsg.innerText=msg; chatBox.appendChild(userMsg); document.getElementById("messageInput").value=""; const response = await fetch("/chat", { method:"POST", headers:{"Content-Type":"application/json"}, body: JSON.stringify({ message: msg, guests: parseInt(document.getElementById("guestInput").value), budget: parseInt(document.getElementById("budgetInput").value), style: document.getElementById("styleSelect").value }) }); const data = await response.json(); const aiMsg = document.createElement("div"); aiMsg.className="message ai"; aiMsg.innerText=data.reply; chatBox.appendChild(aiMsg); chatBox.scrollTop = chatBox.scrollHeight; // 如果回傳 timeline 自動更新圖表 if(data.timeline) updateTimelineChart(data.timeline); // 如果回傳預算，自動更新圖表 if(data.budget) updateBudgetChart(document.getElementById("budgetInput").value); } // 流程時間軸 + tooltip function updateTimelineChart(timelineData){ const labels = timelineData.map(item => item.step); const durations = timelineData.map(item => item.duration); const suggestions = timelineData.map(item => item.suggestion || "建議: 按標準流程安排"); const ctx = document.getElementById('timelineChart').getContext('2d'); if(chartTimeline) chartTimeline.destroy(); chartTimeline = new Chart(ctx,{ type:'bar', data:{ labels: labels, datasets:[{ label:'時間(小時)', data:durations, backgroundColor:['#1d2671','#FF6384','#36A2EB','#FFCE56'] }] }, options:{ indexAxis:'y', responsive:true, plugins:{ legend:{display:false}, tooltip:{ callbacks:{ label: function(context){ const idx = context.dataIndex; return `時間: ${durations[idx]} 小時 | ${suggestions[idx]}`; } } } }, scales:{ x:{beginAtZero:true,title:{display:true,text:'小時'}}, y:{title:{display:true,text:'流程'}} } } }); }
